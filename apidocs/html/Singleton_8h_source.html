00049 /// Mix-in class implementing the Singleton pattern. A Singleton
00050 /// object is never created or destroyed explicitly. Instead, the
00051 /// object is created when the Instance() method is called the first
00052 /// time. The object is destroy when DestroyInstance() is called.
00053 ///
00054 template <class singleton_t>
00055 class MeaSingleton_T
00056 {
00057 public:
00058     /// Destructor for the Singleton. This method is called
00059     /// indirectly by the DestroyInstance() method. The delete
00060     /// operator is never directly applied to a Singleton object
00061     /// by the user of this class.
00062     ///
00063     virtual ~MeaSingleton_T() { }
00064 
00065 
00066     /// The first time this method is called, a instance of the
00067     /// Singleton object is created and returned. Subsequent calls
00068     /// to this method return the same object instance. The method
00069     /// asserts if DestroyInstance() has been called.
00070     ///
00071     /// @return Singleton instance. The same instance is always
00072     ///         returned.
00073     ///
00074     static singleton_t& Instance() {
00075         MeaAssert(!m_destroyed);
00076         if (m_instance == NULL) {
00077             m_instance = new singleton_t();
00078         }
00079         return *m_instance;
00080     }
00081 
00082     /// When the Singleton object is no longer needed, it must
00083     /// be destroyed using this method. The Instance() method must
00084     /// not be called after this method has been called or an
00085     /// assertion will be raised.
00086     ///
00087     static void DestroyInstance() {
00088         delete m_instance;
00089         m_instance = NULL;
00090         m_destroyed = true;
00091     }
00092 
00093     /// Indicates whether the Singleton object has been destroyed.
00094     /// Once a Singleton instance has been destroyed, it is illegal
00095     /// to call the Instance() method.
00096     ///
00097     /// @return <b>true</b> if the Singleton object has been destroyed.
00098     ///
00099     static bool IsDestroyed() { return m_destroyed; }
