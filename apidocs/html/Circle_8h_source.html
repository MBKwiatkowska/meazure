00150     /// The circular window is composed of a series of single pixel
00151     /// rectangles arranged in a circle. At each point on the circle
00152     /// the coordinates of a rectangle surrounding that point are added
00153     /// to an array of points. This array of points will eventually be
00154     /// handed to the GDI function CreatePolyPolygonRgn.
00155     ///
00156     /// @param x    [in] X coordinate of the point on the circle
00157     /// @param y    [in] Y coordinate of the point on the circle
00158     ///
00159     void    AddPoint(int x, int y) {
00160         POINT p;
00161 
00162         // Put the coordinates into a point
00163         //
00164         p.x = x;
00165         p.y = y;
00166 
00167         // Make sure the point is somewhere on the virtual rectangle
00168         // formed by all display monitors.
00169         //
00170         if (m_clipRect.PtInRect(p)) {
00171             // Determine where in the coordinate array
00172             // we will start adding the rectangle coordinates.
00173             //
00174             register POINT *arr = &m_arr[m_count << 2];
00175 
00176             // (x,y)   (x+1, y)
00177             //    ---------
00178             //   |         |
00179             //   |         |
00180             //   |         |
00181             //    ---------
00182             // (x,y+1) (x+1,y+1)
00183 
00184 
00185             // (x,y)
00186             //
00187             *arr++ = p;
00188 
00189             // (x+1,y)
00190             //
00191             p.x++;
00192             *arr++ = p;
00193 
00194             // (x+1,y+1)
00195             //
00196             p.y++;
00197             *arr++ = p;
00198 
00199             // (x, y+1)
00200             //
00201             p.x--;
00202             *arr = p;
00203 
00204             // Keep track of the number of rectangles
00205             //
00206             m_count++;
00207         }
00208     }
